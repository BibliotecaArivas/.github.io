CREATE DATABASE Academico;
USE Academico;

-- Esquema 
CREATE SCHEMA Academico


-- Tabla Cursos 
CREATE TABLE Cursos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre TEXT,
    duracion TIME NOT NULL
);

-- Actualizar TIME
UPDATE Cursos SET duracion = '02:00:00' WHERE id = 1;

DROP TABLE IF EXISTS Cursos;
